from numpy import *
import matplotlib.pyplot as plt

data = genfromtxt('spots.txt')

Zadanie 2 - proste filtrowanie

Wykrel sygna $sin(2*pi*t) + sin(4*pi*t)$, T=1s, w=20Hz. Za pomoc FFT, przekszta sygna do dziedziny czstotliwoci. Nastpnie usu skadowe o czstotliwoci 2Hz. Tak zmodyfikowany sygna przekszta do dziedziny czasu i wykrel go.

LP = 1.0
T = 1.0         
w = 20.0         
TW = 1.0 / w     
t = arange(0, LP*T, TW)  
n = len(t)    

FUNC = lambda t: sin(2*pi*t) + sin(4*pi*t)
signal = FUNC(t)

base_t = arange(0, LP*T, 1.0/200.0)
base_signal = FUNC(base_t)
 
freqs = linspace(0, w*(n-1)/n, n)
signal_f = np.fft.fft(signal)
abs_signal_f = np.abs(signal_f)/(n/2) 

# Usunicie skadowej o czstotliwoci 2Hz
# Znajd indeksy czstotliwoci 2Hz (bliskie 2Hz z tolerancj 0.1)
# to_filter = [2] 
to_filter = where(isclose(freqs, 2, atol=0.1))[0]  
filtered_signal_f = copy(signal_f)
filtered_signal_f[to_filter] = 0 

filtered_signal = real(fft.ifft(filtered_signal_f))  # Przeksztacenie do dziedziny czasu

plt.figure(figsize=(15, 6), dpi=80)

Zadanie 3 - Informacja o fazie

Wykrel sygna $sin(2*pi*t) + sin(4*pi*t)$, T=1s, w=20Hz. Tym razem oprcz spektrum, wykrel wykres z informacj o fazie poszczeglnych czstotliwoci (faza = arg(z)), gdzie z=a+bi. To samo wykonaj dla sin(2pit) + cos(4pit). Porwnaj otrzymane wykresy. Przydatna moe by funkcja angle (z parametrem deg=True)

In[105]:
def func_1(t):
    return sin(2*pi*t) + sin(4*pi*t)

def func_2(t):
    return sin(2*pi*t) + cos(4*pi*t)
    
LP = 1.0
T = 1.0         
w = 20.0         
TW = 1.0 / w     
t = arange(0, LP*T, TW)  
n = len(t)    

funcs = [func_1, func_2]

for func in funcs:
    signal = func(t)

    base_t = arange(0, LP*T, 1.0/200.0)
    base_signal = func(base_t)

    freqs = linspace(0, w*(n-1)/n, n)
    signal_f = np.fft.fft(signal)
    abs_signal_f = np.abs(signal_f)/(n/2)
    
    #Faza kadej skadowej czstotliwoci sygnau po przeksztacenia Fouriera.
    phase = angle(signal_f, deg=True)

    plt.figure(figsize=(20, 6), dpi=80)
    
    plt.subplot(1, 3, 1)
    plt.title('Sygna w dziedzinie czasu')
    plt.ylabel('Amplituda')
    plt.xlabel('t [s]')
    plt.ylim(-2.2, 2.2)
    plt.plot(t, signal, 'o')
    plt.plot(base_t, base_signal, color='blue')
    
    plt.subplot(1, 3, 2)
    plt.title('Dziedzina czstotliwoci')
    plt.stem(freqs, abs_signal_f)
    plt.xlabel('Czstotliwo [Hz]')
    plt.ylabel('Amplituda')
    
    plt.subplot(1, 3, 3)
    plt.title('Faza (Hz)')
    plt.stem(freqs, phase)
    plt.yticks(range(-180, 181, 45))
    plt.xlabel('Czstotliwo [Hz]')
    plt.ylabel('Faza (stopnie)')
    
    plt.show()
